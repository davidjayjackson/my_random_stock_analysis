Box Plot Of Closing Prices

plot_monthly_boxplot <- function(ticker, start_date = "2024-01-01") {
  
  # Download data
  getSymbols(ticker, from = start_date, auto.assign = TRUE)
  
  # Prepare tidy data
  df <- get(ticker) %>%
    fortify.zoo() %>%
    as_tibble() %>%
    rename(Date = Index, Close = paste0(ticker, ".Close")) %>%
    mutate(
      Month = floor_date(Date, "month"),
      MonthLabel = format(Month, "%b %Y"),
      MonthLabel = factor(MonthLabel, levels = unique(MonthLabel))
    )
  
  # Compute monthly mean
  monthly_mean <- df %>%
    group_by(MonthLabel) %>%
    summarize(MeanClose = mean(Close, na.rm = TRUE))
  
  # Plot
  ggplot(df, aes(x = MonthLabel, y = Close)) +
    geom_boxplot(fill = "skyblue", color = "gray20", outlier.alpha = 0.4) +
    geom_point(
      data = monthly_mean,
      aes(x = MonthLabel, y = MeanClose),
      color = "orange", size = 3
    ) +
    labs(
      title = paste("Monthly Closing Price Distribution for", ticker),
      subtitle = paste("From", start_date, "to", format(max(df$Date), "%Y-%m-%d")),
      x = "Month",
      y = "Closing Price ($)"
    ) +
    theme_minimal() +
    theme(axis.text.x = element_text(angle = 45, hjust = 1))
}

# ðŸ§© Example usage
plot_monthly_boxplot("IBM", start_date = "2025-06-01")

df <- get(ticker) %>%
  fortify.zoo() %>%
  as_tibble() %>%
  rename(Date = Index, Close = paste0(ticker, ".Close")) %>%
  mutate(Month = floor_date(Date, "month"))

# Build two points per month: first & last close
month_points <- df %>%
  group_by(Month) %>%
  arrange(Date, .by_group = TRUE) %>%
  reframe(
    Date  = c(first(Date),  last(Date)),
    Close = c(first(Close), last(Close)),
    Type  = c("First", "Last")
  ) %>%
  ungroup()

# One line per month connecting first â†” last close, with point markers
ggplot(month_points, aes(x = Date, y = Close, group = Month, color = factor(Month))) +
  geom_line(size = 1) +
  geom_point(aes(shape = Type), size = 3) +
  scale_shape_manual(values = c(First = 16, Last = 17), name = "Point") +
  labs(
    title = paste("Monthly First â†” Last Closing Prices for", ticker),
    subtitle = "Each colored line connects the first and last close in a month",
    x = "Date",
    y = "Closing Price ($)",
    color = "Month"
  ) +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))
